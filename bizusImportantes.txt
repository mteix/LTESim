#### Awk filtering
grep sinr mob_sinrplot_randdir | awk '{if ((NR+6)%6==4) print NR,$2}' > sinr4All.dat

### Awk copying last records from a file 
grep sinr mob_sinrplot_randdir | awk '{if ((NR+6)%6==4) {for (i=3;i<=NF;i++) printf "%s ", $i;print ""}}' > sinr4All.dat
